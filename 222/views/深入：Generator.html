Symbol.iterator 代表的是迭代器函数

  1. Symbol.iterator 中的return：返回的就是一个迭代器，迭代器是一个对象，对象上需要有一个next方法

  2. Symbol.iterator return 中的next()会默认调用next方法，next方法默认返回一个对象，对象需要包含两个属性，一个是value，另一个叫done

应用：

koa1.0 generator 被async+await干掉了

function sum(){

    //let arr=[...arguments];

    let arr=[...{0:1,1:2,2:3,length:3,[Symbol.iterator]:function(){

    return {

        next(){

            return {done:false,value:123}//注意这里done如果一直是false会奔溃

        }

    }

}}]

    console.log(arr);//[1,2,3,4,5,6]

}

sum(1,2,3,4,5,6);

    let len=0;

        next(){

            return {done:len==that.length,value:that[len++]}

}}]

}

  1. 执行yield 的返回值

next的参数是yield的返回值，第一个next传递参数是没有意义的

function * read(){

    let a=yield 1;

    console.log(a,'111');

    let b=yield 2;

    console.log(b);

    return 3;

}

let it=read();

let {value,done}=it.next();

console.log(value,done);

it.next('hello');

it.next('world');

  1. 使用co库

使用promises为nodejs和浏览器提供基于生成器的控制流优度，让你以一种很好的方式编写非阻塞代码。

let fs=require('mz/fs');

function *read(){

    let content=yield fs.readFile('name.txt','utf8');

    let r=yield fs.readFile(content,'utf8');

    return r;

}

let co=require('co');

co(read()).then(function(data){

    console.log(data);

})
