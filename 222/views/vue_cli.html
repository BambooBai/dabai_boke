Vue：cli

安装vuecli命令并使用

  • sudo npm install @vue/cli -g vue脚手架

  • sudo npm install -g @vue/cli-service-global vuecli运行依赖

    • 可升级

    • 基于 webpack 构建，并带有合理的默认配置

    • 可以通过项目内的配置文件进行配置

    • 可以通过插件进行扩展

  • vue create firstProject 创建vue项目 链接

  • npm run serve 运行项目

  • vue serve Home.vue 运行vue项目

一、入口

main.js

import Vue from 'vue'

import App from './App.vue'



Vue.config.productionTip = false



new Vue({

  render: h => h(App),

}).$mount('#app')

三、vue配置

手动创建vue.config.js

注意：

  • 文件时基于node，node不支持es6的import

  • 默认环境变量 NODE_ENV production development

打包参考链接：链接

let path = require('path');

module.exports = {

    // baseURL

    publicPath:process.env.NODE_ENV === 'production'? 'http://www.zhufeng.cn':'/',

    assetsDir:'asserts',

    outputDir:'./dist',

    // 使用模版方式 一般不使用

    runtimeCompiler:false,

    // 打包 不再使用sourcemap

    productionSourceMap:false,

    chainWebpack:config=>{

        // 可以获取到webpack的配置 在增加一些自己的逻辑

        // 配置目录别名 别名叫+

        config.resolve.alias.set('_c',path.resolve(__dirname,'src/components'));

        config.resolve.alias.set('_v',path.resolve(__dirname,'src/views'));

    },

    configureWebpack:{ // webpack-merge

        plugins:[],

        module:{}

    },

    devServer:{ // 开发 服务时使用

        proxy:{

            '/api/getUser':{

                target:'http://localhost:3000',

                pathRewrite:{

                    '/api':''

                }

            }

        }

    },

    pluginOptions: {

      'style-resources-loader': { //为每个文件都引入这个样式

        preProcessor: 'less',

        patterns: [

            path.resolve(__dirname,'src/assets/common.less')

        ]

      }

    }

}
